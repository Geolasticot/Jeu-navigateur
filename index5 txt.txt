    <script>
        const rooms = {
            library: {
                title: "Bibliothèque",
                description: "Vous êtes entouré de livres anciens remplis de savoir magique.",
            },
            dungeon: {
                title: "Cachots",
                description: "Vous descendez dans les sombres cachots où des secrets sont enfouis.",
            },
            greenhouse: {
                title: "Serre",
                description: "Une serre luxuriante remplie de plantes magiques lumineuses.",
            },
            portal: {
                title: "Portail",
                description: "Un portail scintillant qui mène au monde extérieur.",
            }
        };



     // Ajout des images spécifiques pour chaque pièce
// Ajout des images spécifiques pour chaque pièce
const roomImages = {
    library: 'https://i.imgur.com/MHDCqqE.png',
    dungeon: 'https://i.imgur.com/Pt9vCmy.png',
    greenhouse: 'https://i.imgur.com/okil1HE.png',
    portal: 'https://i.imgur.com/mS5ZfCg.png'
};

// Mettre à jour l'affichage de la pièce avec l'image correspondante
document.querySelectorAll('.interactive-zone').forEach(zone => {
    zone.addEventListener('click', function() {
        const roomId = this.id;
        const roomData = rooms[roomId];
        const roomImage = roomImages[roomId]; // Récupérer l'image associée à la pièce

        // Afficher les détails de la pièce
        if (roomData) {
            document.getElementById('room-title').innerText = roomData.title;
            document.getElementById('room-description').innerText = roomData.description;

            // Ajouter l'image de la pièce comme arrière-plan complet
            if (roomImage) {
                document.getElementById('room-container').style.backgroundImage = `url('${roomImage}')`;
            }

            // Basculer l'affichage
            document.getElementById('map-container').style.display = 'none';
            document.getElementById('message-box').style.display = 'none';
            document.getElementById('room-container').style.display = 'block';
        }
    });
});


        // Gestion du bouton Retour
        document.getElementById('back-button').addEventListener('click', function() {
            // Basculer l'affichage vers la carte
            document.getElementById('map-container').style.display = 'block';
            document.getElementById('message-box').style.display = 'block';
            document.getElementById('room-container').style.display = 'none';
        });
    </script>
